// ━━━ kernel/paul_hw.cpp ━━━

#include <ros/ros.h>
#include <Arduino.h>

class Paul : public Engineer {
private:
  ROS::Node    node_;
  GPIO::Pin    sensors_[16];
  Serial       uart_;

public:
  void boot() {
    node_.init("paul_core");
    for (auto& s : sensors_)
      s.attach(INPUT);
    ROS_INFO(">> HW ONLINE");
  }

  void execute() {
    while (ros::ok()) {
      navigate_world();
      avoid_obstacles();
      ros::spinOnce();
    }
  }
};
